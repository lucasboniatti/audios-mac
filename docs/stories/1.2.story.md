### Transcription.swift

```swift
import CoreData
import Foundation

@objc(Transcription)
public class Transcription: NSManagedObject {
    @NSManaged public var id: UUID
    @NSManaged public var text: String
    @NSManaged public var timestamp: Date
}

extension Transcription {
    static func fetchRequest() -> NSFetchRequest<Transcription> {
        return NSFetchRequest<Transcription>(entityName: "Transcription")
    }
}
```

### PersistenceService.swift

```swift
import CoreData
import Foundation

class PersistenceService {
    static let shared = PersistenceService()
    
    let container: NSPersistentContainer
    
    init() {
        container = NSPersistentContainer(name: "AudioFlow")
        container.loadPersistentStores { _, error in
            if let error = error {
                fatalError("CoreData error: \(error)")
            }
        }
    }
    
    var viewContext: NSManagedObjectContext {
        container.viewContext
    }
    
    func save() {
        let context = viewContext
        if context.hasChanges {
            try? context.save()
        }
    }
}
```
