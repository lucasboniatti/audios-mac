## Technical Details

### Search Implementation

```swift
extension HistoryController {
    func searchTranscriptions(query: String) -> [Transcription] {
        let request: NSFetchRequest<Transcription> = Transcription.fetchRequest()
        request.sortDescriptors = [NSSortDescriptor(key: "timestamp", ascending: false)]
        request.predicate = NSPredicate(format: "text CONTAINS[cd] %@", query)
        
        return (try? PersistenceService.shared.viewContext.fetch(request)) ?? []
    }
}
```

### UI Approach

Como não temos janela principal, usar **NSPanel** para busca:

```swift
class SearchPanelController: NSWindowController {
    var searchField: NSSearchField!
    var tableView: NSTableView!
    var results: [Transcription] = []
    
    func search(_ query: String) {
        results = HistoryController.shared.searchTranscriptions(query: query)
        tableView.reloadData()
    }
}
```

---

## Dev Notes

- `CONTAINS[cd]` no predicate: case-insensitive e diacritic-insensitive
- NSPanel é uma janela flutuante que não aparece no Dock
- Alternativa: mostrar busca dentro do menu (mais complexo)
