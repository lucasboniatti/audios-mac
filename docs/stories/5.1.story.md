## Technical Details

### Export Format

```
[2026-03-05 14:30:00] Primeira transcrição aqui...
[2026-03-05 14:35:00] Segunda transcrição aqui...
```

### Implementation

```swift
extension HistoryController {
    func exportToTXT(to url: URL) throws {
        let transcriptions = getRecentTranscriptions(limit: 100)
        let dateFormatter = DateFormatter()
        dateFormatter.dateFormat = "yyyy-MM-dd HH:mm:ss"
        
        var content = ""
        for t in transcriptions {
            let timestamp = dateFormatter.string(from: t.timestamp)
            content += "[\(timestamp)] \(t.text)\n\n"
        }
        
        try content.write(to: url, atomically: true, encoding: .utf8)
    }
}
```
