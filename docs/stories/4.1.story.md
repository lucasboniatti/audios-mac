### HistoryController.swift

```swift
import CoreData

class HistoryController {
    static let shared = HistoryController()
    
    private let maxItems = 10
    
    func saveTranscription(_ text: String) {
        let context = PersistenceService.shared.viewContext
        let transcription = Transcription(context: context)
        transcription.id = UUID()
        transcription.text = text
        transcription.timestamp = Date()
        
        PersistenceService.shared.save()
        
        enforceFIFOLimit()
    }
    
    func getRecentTranscriptions(limit: Int = 10) -> [Transcription] {
        let request: NSFetchRequest<Transcription> = Transcription.fetchRequest()
        request.sortDescriptors = [NSSortDescriptor(key: "timestamp", ascending: false)]
        request.fetchLimit = limit
        
        return (try? PersistenceService.shared.viewContext.fetch(request)) ?? []
    }
    
    private func enforceFIFOLimit() {
        let all = getRecentTranscriptions(limit: 100)
        if all.count > maxItems {
            let toRemove = all.suffix(from: maxItems)
            for item in toRemove {
                PersistenceService.shared.viewContext.delete(item)
            }
            PersistenceService.shared.save()
        }
    }
}
```

### Menu Integration

```swift
// No MenuBuilder
func buildHistoryMenu() -> NSMenu {
    let history = HistoryController.shared.getRecentTranscriptions()
    
    let menu = NSMenu()
    for transcription in history {
        let preview = String(transcription.text.prefix(30)) + "..."
        let item = NSMenuItem(title: preview, action: #selector(copyTranscription(_:)), keyEquivalent: "")
        item.representedObject = transcription
        menu.addItem(item)
    }
    return menu
}
```
